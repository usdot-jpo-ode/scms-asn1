58 -- between Misbehavior Authority (MA) and Pseudonym Certificate

61 -- @param maPcaPreLinkageValueRequest contains the PLV request message sent

78 -- @brief MA uses this data type to request pre-linkage value from PCA for
79 -- misbehavior detection. Note that this is actually signed by the MA
80 -- at the scms-protocol level.
81 -- @class MaPcaPreLinkageValueRequestMsg
82 -- @param version contains the current version of the data type. The version
83 -- specified in this document is version 1, represented by
84 -- the integer 1.
85 -- @param tbs is the request message data that is signed by the MA at
86 -- scms-protocol level.
87 -- @param signatures contains countersignatures that the MA has to obtain from
88 -- auditors before requesting PLVs from PCA. Size 0 is used
89 -- when no countersignatures are necessary.
90 -- @see Uint8, Countersignature
91 MaPcaPreLinkageValueRequestMsg ::= SEQUENCE {
92  version Uint8(1),
93  tbs ToBeSignedMaPcaPreLinkageValueRequestMsg,
94  signatures SEQUENCE (SIZE (0..MAX)) OF Countersignature
95 }
