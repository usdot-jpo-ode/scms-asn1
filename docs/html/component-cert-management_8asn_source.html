71 ---
72 -- @brief An "endorsement" is a structure with a command type (add/remove for
73 -- root or elector) along with the certificate to be added or removed.
74 -- A signed endorsement is a copy of that whole structure with a
75 -- signature.
76 -- @class TbsElectorEndorsement
77 -- @param type defines the action (add/remove) to be taken over a
78 -- certificate.
79 -- @param certificate denotes the certificate to be added or removed.
80 -- @param effectiveTime defines the effective time for this message.
81 -- @see ExplicitCertificate, Time64
82 TbsElectorEndorsement ::= SEQUENCE {
83  type EndorsementType,
84  certificate ExplicitCertificate,
85  effectiveTime Time64 OPTIONAL
86 }

88 ---
89 -- @brief This data type defines the action (add/remove) to be taken on a Root
90 -- CA or Elector certificate.
91 -- @class EndorsementType
92 -- @param addRoot defines the action to add a Root CA.
93 -- @param addElector defines the action to add an elector.
94 -- @param removeRoot defines the action to remove a Root CA.
95 -- @param removeElector defines the action to remove an elector.
96 EndorsementType ::= ENUMERATED {
97  addRoot (0),
98  addElector (1),
99  removeRoot (2),
100  removeElector (3),
101  ...
102 }

104 ---
105 -- @brief This data type defines the structure of an elector ballot. A "ballot"
106 -- is a list of signed endorsements. To be valid, a ballot must contain
107 -- signatures from a quorum of electors.
108 -- @class ElectorBallot
109 -- @param endorsements are the signatures are generated by the Electors (i.e.
110 -- endorsement ballots). Each signature shall contain a
111 -- copy of the same TbsElectorEndorsement
112 -- @see SignedElectorEndorsement
113 ElectorBallot ::= SEQUENCE {
114  -- ballot TbsBallot,
115  endorsements SEQUENCE SIZE(1..MAX) OF SignedElectorEndorsement,
116  ...
117 }

120 -- @brief This data type defines a composite CRL structure which contains a
121 -- list of 1609.2 CRLs, one per CRL series. It may also contain ballots
122 -- to remove Root CAs and Electors
123 -- @class CompositeCrl
124 -- @param securedCrlSeries is the list of CRLs for all SCMS components except
125 -- a Root CA and Electors.
126 -- @param revokedRootCAs is the list of Elector endorsements that were used

128 -- @param revokedElectors is the list of Elector endorsements that were used

130 -- @see SecuredCrl

132  securedCrlSeries SEQUENCE SIZE(0..MAX) OF SecuredCrl,
