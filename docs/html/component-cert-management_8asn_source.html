74 ---
75 -- @brief An "endorsement" is a structure with a command type (add/remove for
76 -- root or elector) along with the certificate to be added or removed.
77 -- A signed endorsement is a copy of that whole structure with a
78 -- signature.
79 -- @class TbsElectorEndorsement
80 -- @param type Defines the action (add/remove) to be taken over a
81 -- certificate.
82 -- @param certificateHash The hash of the certificate with respect to which the
83 -- action is to be taken.
84 -- @param generationTime Defines the time at which the message was generated.
85 -- The message takes effect immediately on generation.
86 -- For an add action type, the added certificate is considered
87 -- to be valid from the start of its validity period, no matter
88 -- what the generation time value in this message is. For a
89 -- revoke action type, the revoked certificate is considered
90 -- revoked from the time of the earliest generation time on
91 -- a valid revoke message.
92 -- @see HashedId32
93 TbsElectorEndorsement ::= SEQUENCE {
94  type EndorsementType,
95  certificateHash HashedId32,
96  generationTime Time32,
97  ...
98 }
99 
100 ---
101 -- @brief This data type defines the action (add/remove) to be taken on a Root
102 -- CA or Elector certificate.
103 -- @class EndorsementType
104 -- @param addRoot defines the action to add a Root CA.
105 -- @param addElector defines the action to add an elector.
106 -- @param removeRoot defines the action to remove a Root CA.
107 -- @param removeElector defines the action to remove an elector.
108 EndorsementType ::= ENUMERATED {
109  addRoot (0),
110  addElector (1),
111  removeRoot (2),
112  removeElector (3),
113  ...
114 }

116 ---
117 -- @brief This data type defines the structure of an elector ballot. A "ballot"
118 -- is a list of signed endorsements. To be valid, a ballot must contain
119 -- signatures from a quorum of electors.
120 -- @class ElectorBallot
121 -- @param endorsements are the signatures are generated by the Electors (i.e.
122 -- endorsement ballots). Each signature shall contain a
123 -- copy of the same TbsElectorEndorsement
124 -- @see SignedElectorEndorsement
125 ElectorBallot ::= SEQUENCE {
126  -- ballot TbsBallot,
127  -- each signature shall contain a copy of the same TbsElectorEndorsement
128  certificate ExplicitCertificate, -- root CA or elector
129  endorsements SEQUENCE SIZE(1..MAX) OF SignedElectorEndorsement,
130  ...
131 }

133 ---
134 -- @brief This data type defines a composite CRL structure which contains a
135 -- list of 1609.2 CRLs, one per CRL series. It may also contain ballots
136 -- to remove Root CAs and Electors
137 -- @class CompositeCrl
138 -- @param securedCrlSeries is the list of CRLs for all SCMS components except
139 -- a Root CA and Electors.
140 -- @param revokedRootCAs is the list of Elector endorsements that were used
141 -- to revoke a Root CA.
142 -- @param revokedElectors is the list of Elector endorsements that were used
143 -- to revoke an Elector.
144 -- @see SecuredCrl
145 CompositeCrl ::= SEQUENCE {
146  securedCrlSeries SEQUENCE SIZE(0..MAX) OF SecuredCrl,
147  revokedRootCAs SEQUENCE SIZE(0..MAX) OF ElectorBallot,
148  revokedElectors SEQUENCE SIZE(0..MAX) OF ElectorBallot,
149  ...
150 }
