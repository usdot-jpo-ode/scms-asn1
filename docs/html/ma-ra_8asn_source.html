62 -- @brief The MaRaInterfacePDU is the parent message type for messages sent
63 -- between Misbehavior Authority (MA) and Registration Authority (RA).
64 -- @class MaRaInterfacePDU
65 -- @param maRaBlacklistRequest contains request from MA to obtain
66 -- internal blacklist of RA for pseudonym
67 -- certificate.
68 -- @param raMaBlacklistResponse contains response from RA to
69 -- MaRaBlacklistRequestMsg from MA.
70 -- @param maRaLCIRequest contains request from MA to obtain
71 -- linkage chain ID from RA.
72 -- @param raMaLCIResponse contains response from RA to
73 -- MaRaLCIRequestMsg from MA.
74 -- @param maRaRseObeIdBlacklistRequest contains request from MA to obtain
75 -- internal blacklist of RA for all
76 -- certificates expect pseudonym
77 -- certificate.
78 -- @param raMaRseObeIdBlacklistResponse contains response from RA to
79 -- MaRaRseObeIdBlacklistRequest from MA.
80 -- @param maRaCDVRequest contains request from MA to obtain
81 -- certificate digest values (CDV) from RA.
82 -- @param raMaCDVResponse contains response from RA to
83 -- MaRaCDVRequest from MA.
84 MaRaInterfacePDU ::= CHOICE {
85  maRaBlacklistRequest MaRaBlacklistRequestMsg,
86  raMaBlacklistResponse RaMaBlacklistResponseMsg,
87  maRaLCIRequest MaRaLCIRequestMsg,
88  raMaLCIResponse RaMaLCIResponseMsg,
89  maRaRseObeIdBlacklistRequest MaRaRseObeIdBlacklistRequestMsg,
90  raMaRseObeIdBlacklistResponse RaMaRseObeIdBlacklistResponseMsg,
91  maRaCDVRequest MaRaCDVRequestMsg,
92  raMaCDVResponse RaMaCDVResponseMsg,
93  ...
94 }

97 -- @brief This data type is used by MA to request internal blacklist of RA.
98 -- Note that this is actually signed by the MA at the scms-protocol
99 -- level.
100 -- @class MaRaBlacklistRequestMsg
101 -- @param version contains the current version of the data type. The version
102 -- specified in this document is version 1, represented by
103 -- the integer 1.
104 -- @param tbs contains information of MA and requested HPCR. MA signs
105 -- this at scms-protocol level.
106 -- @param signatures is the list of counter signatures that the MA has to
107 -- obtain from Auditors before sending this request. Size 0
108 -- is used when no countersignatures are necessary
109 -- @see Uint8, Countersignature
110 MaRaBlacklistRequestMsg ::= SEQUENCE {
111  version Uint8(1),
112  tbs ToBeSignedBlacklistingInstructionMsg,
113  signatures SEQUENCE SIZE (0..MAX) OF Countersignature
114 }
