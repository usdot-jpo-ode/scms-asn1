93 -- @brief This data type is used by the RA to request pseudonym certificates from the PCA. It is
94 -- signed using the private key generated by the RA and the corresponding public key,
95 -- verificationKey, is placed for use by the PCA to generate the pseudonym certificates.
96 -- @class RaPcaCertRequestMsg
97 -- @param version contains the current version of the data type. The version specified in this
98 -- document is version 1, represented by the integer 1.
99 -- @param raId contains the RA's hostname.
100 -- @param cert-request-info contains the information to request certificates for OBEs. A
101 -- PseudonymCertRequestInfo is included when the OBE requires pseudonimity
102 -- and unlinkability. IdCertRequestInfo is included otherwise.
103 -- @see Uint8, RaHostnameId
104 RaPcaCertRequestMsg ::= SEQUENCE {
105  version Uint8(1),
106  raId RaHostnameId,
107  cert-request-info CHOICE {
108  pseudonym-cert-req PseudonymCertRequestInfo,
109  auth-cert-req IdCertRequestInfo,
110  ...
111  }
112 }

115 -- @brief OBE Pseudonym Certificate info sent in the request certificate message. This version is only
116 -- for implicit certs using butterfly keys. If other versions of pseudonym cert requests are
117 -- developed with no butterfly keys or for explicit certs, that can be handled by defining an
118 -- additional CertRequestInfo type within RaPcaCertRequestMsg. In particular, this doesn't
119 -- allow pseudonym certs to have encryption keys, although it would in principle be possible to
120 -- support this if the CA added a value to the encryption key as well as modifying the signing
121 -- key. It's probably best for pseudonymous devices that want encrypted responses to do it via
122 -- a response encryption key in the SignedData anyway.
123 -- @class PseudonymCertRequestInfo
124 -- @param psidSsp is a sequence of PSIDs with corresponding SSPs.
125 -- @param performanceAssuranceLevel indicates the assurance level of the certificate holder. This field is optional.
126 -- @param region defines the GeographicRegion for which a certificate is being
127 -- requested. This field is optional.
128 -- @param iValue denotes the week number from i=0 (i.e. 4:00 am Eastern Time on
129 -- Tuesday, January 6, 2015)
130 -- @param pca-id contains the PCA's hostname.
131 -- @param sig-butterfly-key-B used to generate the hash of the pseudonym certificate request
132 -- from RA.
133 -- @param la1-id contains first LA's hostname (LA1).
134 -- @param la2-id contains second LA's hostname (LA2).
135 -- @param enc-plv1 contains encrypted pre-linkage value received from LA1 for an
136 -- individual OBE.
137 -- @param enc-plv2 contains encrypted pre-linkage value received from LA2 for an
138 -- individual OBE.
139 -- @param group-linkage-info DEPRECATED: is a sequence containing linkage value information
140 -- of a group of OBEs. This field is optional.
141 -- @param enc-butterfly-key-H contains RA's public key that is used by PCA to encrypt its responses.
142 -- @see SequenceOfPsidSsp, SubjectAssurance, GeographicRegion, IValue
143 PseudonymCertRequestInfo ::= SEQUENCE {
144  --
145  -- these fields go directly into the cert
146  --
147  psidSsp SequenceOfPsidSsp,
148  performanceAssuranceLevel SubjectAssurance OPTIONAL,
149  region GeographicRegion OPTIONAL,
150  iValue IValue,
151  --
152  -- used to create material that goes in the cert but not included in the
153  -- cert directly
154  --
155  pca-id PcaHostnameId,
156  sig-butterfly-key-B EccP256CurvePoint,
157  la1-id LaHostnameId,
158  la2-id LaHostnameId,
159  enc-plv1 EncryptedIndividualPLV,
160  enc-plv2 EncryptedIndividualPLV,
161  group-linkage-info SEQUENCE {
162  gla1-id LaHostnameId,
163  gla2-id LaHostnameId,
164  enc-gplv1 EncryptedGroupPLV,
165  enc-gplv2 EncryptedGroupPLV
166  } OPTIONAL,
167  -- used to encrypt the response
168  enc-butterfly-key-H EccP256CurvePoint
169 }

226 -- @brief SignedEncrypted responses are signed 1609.2 data, containing encrypted 1609.2 data, which
227 -- decrypts to a DecryptedCertificateData containing the PlaintextCertificateResponse.
228 --
229 -- This is implemented as follows.
230 
231 -- At the Signed level:
232 -- * ToBeSignedEncryptedCertificateResponse is defined with all the fields relevant to both
233 -- types of response plus optional i-value
234 -- * SignedEncryptedCertificateResponse is defined.
235 
236 -- At the Encrypted level, the EncryptedCertificateData type is a 1609.2 data containing a
237 -- single recipient info which is encrypted to a response encryption key.
238 
239 -- The EncryptedCertificateData decrypts to a DecryptedCertificateData, which is an unsecured
240 -- 1609.2 data containing a PlaintextCertificateResponse. (It's a 1609.2 data to maintain
241 -- the convention that all EncryptedData decrypts to a 1609.2 data).
242 
243 -- The PlaintextCertificateResponse distinguishes between: (implicit/explicit) and
244 -- (butterfly/single-key) as all four combinations are potentially legal and a receiver needs
245 -- to know which combination applies in order to recover the correct private key.
246 -- @class SignedEncryptedCertificateResponse
247 -- @param unsecuredData contains the response that is signed by the PCA.
248 SignedEncryptedCertificateResponse ::= Ieee1609Dot2Data (WITH COMPONENTS {...,
249  content (WITH COMPONENTS {
250  signedData (WITH COMPONENTS {...,
251  tbsData (WITH COMPONENTS {
252  payload (WITH COMPONENTS {...,
253  data (WITH COMPONENTS {...,
254  content (WITH COMPONENTS {
255  unsecuredData (CONTAINING ToBeSignedEncryptedCertificateResponse)
256  })
257  }) PRESENT,
258  extDataHash ABSENT
259  }),
260  headerInfo (WITH COMPONENTS {...,
261  psid (SecurityMgmtPsid),
262  generationTime ABSENT,
263  expiryTime ABSENT,
264  generationLocation ABSENT,
265  p2pcdLearningRequest ABSENT,
266  missingCrlIdentifier ABSENT,
267  encryptionKey ABSENT
268  })
269  })
270  })
271  })
272 })
