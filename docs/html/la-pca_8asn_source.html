59 -- between Linkage Authority (LA) and Pseudonym Certificate Authority
60 -- (PCA).
61 -- @class LaPcaInterfacePDU
62 -- @param pcaLaKeyAgreementRequest contains the key agreement request sent
63 -- from PCA to LA.
64 -- @param laPcaKeyAgreementResponse contains the response to key agreement

69  pcaLaKeyAgreementRequest PcaLaKeyAgreementRequestMsg,

71  pcaLaKeyAgreementAck PcaLaKeyAgreementAckMsg,

76 -- @brief PCA uses this data type to initiate key agreement with an LA.
77 -- @class PcaLaKeyAgreementRequestMsg
78 -- @param version contains the current version of the data type. The
79 -- version specified in this document is version 1,
80 -- represented by the integer 1.
81 -- @param pcaId is the unique hostname ID of a PCA.
82 -- @param pcaCertificate encloses PCA's certificate. This field is optional in
83 -- the case when LA already has PCA's certificate.
84 -- @param enc-R-pca is the encrypted data generated by using PCA's random
85 -- contribution. The encrypted data contains recipient's
86 -- certificate information.
87 -- @param startDate denotes the start date from which the request is
88 -- valid.
89 -- @param endDate denotes the end date until which the request in valid.
90 -- @see Uint8, PcaHostnameId, Certificate, EncryptedData, Time32
91 PcaLaKeyAgreementRequestMsg ::= SEQUENCE {
92  version Uint8(1),
93  pcaId PcaHostnameId,
94  pcaCertificate Certificate OPTIONAL,
95  enc-R-pca EncryptedData (WITH COMPONENTS {
96  recipients (SIZE (1) INTERSECTION (WITH COMPONENT (
97  RecipientInfo (WITH COMPONENTS {
98  certRecipInfo -- using hash of the LA certificate
99  })))),
100  ciphertext -- when decrypted it contains PcaRandomContribution, R_pca
101  }),
102  startDate Time32,
103  endDate Time32
104 }

113 ---
114 -- @brief This data type is used by the LA to respond to a
115 -- PcaLaKeyAgreementRequestMsg from a PCA.
116 -- @class LaPcaKeyAgreementResponseMsg
117 --- @param version contains the current version of the data type. The
118 -- version specified in this document is version 1,
119 -- represented by the integer 1.
120 -- @param requestHash is the hash of the original request.
121 -- @param laId is the unique hostname ID of a LA.
122 -- @param enc-R-la is the encrypted data generated by using LA's random
123 -- contribution. The encrypted data contains recipient's
124 -- certificate information.
125 -- @param reply in case of no errors in PCA's request, LA responds with
126 -- a full hash that is computed as follows:
127 -- MAC_la computed as HMAC(mac_key, R_pca || R_la || "scms-la")
128 -- where (enc_key, mac_key) = KDF(R_pca || R_la || "scms-la-pca")
129 -- enc_key is the negotiated symmetric key to be used by LA
130 -- for PLV encryption. In case of an error, an appropriate
131 -- error code is sent in the response.
132 -- @see Uint8, HashedId8, LaHostnameId, EncryptedData, FullSizeHash,
133 -- ScopedLaPcaError
134 LaPcaKeyAgreementResponseMsg ::= SEQUENCE {
135  version Uint8(1),
136  requestHash HashedId8, -- hash of the original request
137  laId LaHostnameId,
138  enc-R-la EncryptedData (WITH COMPONENTS {
139  recipients (SIZE (1) INTERSECTION (WITH COMPONENT (
140  RecipientInfo (WITH COMPONENTS {
141  certRecipInfo -- using hash of the PCA certificate
142  })))),
143  ciphertext -- when decrypted it contains LaRandomContribution, R_la
144  }),
145  reply CHOICE {
146  success FullSizeHash,
147  failure ScopedLaPcaError
148  }
149 }
